\begin{algorithm}[!h]
\caption{Progressive fast-path and slow-path opaque HyTM implementation that provides uninstrumented writes and invisible reads; code for transaction $T_k$}
\label{alg:inswrite}
\begin{algorithmic}[1]
  	\begin{multicols}{2}
  	{
  	\footnotesize
	\Part{Shared objects}{
		\State $v_j \in \mathbb{D}$, for each t-object $X_j$ 
		\Statex ~~~~~allows reads, writes and cas
		%\State $L[i]$, a multi-trylock object for transaction $T_i$ 
		\State $r_{j} \in \mathbb{M}$, a versioned lock for each t-object $X_j$
		\Statex ~~~~~allows reads, writes and cas
		%\State implemented from reads and writes
		%\State $L$, multi-trylock
	}\EndPart	
	\Statex
	\Part{Process local objects}{
		\State $\Rset(T_k)$ list storing $\{X_j,r_j\}$
		\State $\Wset(T_k)$ bloom filter storing $\{X_j, v_j\}$
	}\EndPart
	\Statex
	
	\textbf{Code for slow-path transactions}
	\Statex
	\Part{\Read$_k(X_j)$}\quad\Comment{slow-path}{
		  \If{$X_j\in \Wset(T_k)$}
		    \Return $\Wset(T_k).\lit{locate}(X_j)$ \EndReturn
		  \Else
		  
		  \State $\textit{ov}_j := \Read(v_j)$ \label{line:read2}
		  \State $\textit{or}_j := \Read(r_j)$ \label{line:readorec}
		  \State $\Rset(T_k) := \Rset(T_k)\cup\{X_j,or_j\}$ \label{line:rset}
		  \If{$\textit{or}_j$ BITWISE\_AND $0000...0001$} \label{line:abort0}	
			\Return $A_k$ \EndReturn
		  \EndIf
		 
		  \If{$\exists X_j \in Rset(T_k)$:$(\textit{or}_j\neq \Read(r_j))$} \label{line:valid}
			\Return $A_k$ \EndReturn
		  \EndIf
		  \EndIf
		  \Return $\textit{ov}_j$ \EndReturn
		
   	 }\EndPart
	\Statex
	\Part{\Write$_k(X_j,v)$}\quad\Comment{slow-path}{
		
			\State $\textit{nv}_j := v$
			\State $\Wset(T_k) := \Wset(T_k)\cup\{X_j,\textit{nv}_j\}$
			\Return $\ok$ \EndReturn
		
   	}\EndPart
	\Statex
	
	%\Statex	
	\Part{\TryC$_k$()}\quad\Comment{slow-path}{
		\If{$\Wset(T_k)= \emptyset$}
			\Return $C_k$ \EndReturn \label{line:return}
		\EndIf
		\State $\lit{acquire}(\Wset(T_k))$\label{line:acq} 
		
		\If{$\exists X_j \in Rset(T_k)$:$(\textit{or}_j\neq \Read(r_j))$} \label{line:abort3}
			\State $\lit{release}( \ms{Wset}(T_k))$ 
			\Return $A_k$ \EndReturn
		\EndIf
		\ForAll{$X_j \in \Wset(T_k)$}
	 		\State  $v_j.\lit{write}(\textit{nv}_j)$ \label{line:write}
			 
	 	\EndFor	
		\State $\lit{release}(\Wset(T_k))$   \label{line:rel}		
   		\Return $C_k$ \EndReturn
   	 }\EndPart		
	 
 	\newpage
 	\Part{Function: $\lit{acquire}(Q$)}{
   		\ForAll{$X_j \in Q$}	
			\State $r_j.\lit{setLock}()$ \label{line:acq1}
			 		
		\EndFor
		
	}\EndPart		
	 \Statex
	 \Part{Function: $\lit{release}(Q)$}{
  		\ForAll{$X_j \in Q$}	
 			\State $r_j.\lit{unLock}()$ \label{line:rel1}
		\EndFor
		
	}\EndPart
	
		
	\Statex
	\textbf{Code for fast-path transactions}
	\Part{$\textit{read}_k(X_j)$}\quad\Comment{fast-path}{
		\State $\textit{ov}_j := \Read(v_j)$ \Comment{cached read} \label{line:lin1}
		\State $\textit{or}_j := \Read(r_j)$
		\If{$\textit{or}_j$ BITWISE\_AND $0000...0001$}  \Comment{Uncached}		\label{line:hread}
			\Return $A_k$ \EndReturn
		\EndIf
		
		\Return $\textit{ov}_j$ \EndReturn
		
   	 }\EndPart
	\Statex
	\Part{$\textit{write}_k(X_j,v)$}{\quad\Comment{fast-path}
		\State $\textit{or}_j L= \Read(r_j)$
		\If{$\textit{or}_j$ BITWISE\_AND $0000...0001$}  \Comment{Uncached}		
			\Return $A_k$ \EndReturn
		\EndIf
		
		\State $\Write(v_j,v)$ \Comment{cached write} 
		\State $\Write(r_j,\textit{or}_j+2)$ \Comment{uncached write}
		%\State $\ms{Increment}(c_j)$ \Comment{Uncached write}
		\Return $\ok$ \EndReturn
		
   	}\EndPart
	\Statex
	
	\Part{$\textit{tryC}_k$()}{\quad\Comment{fast-path}
		\State $\ms{commit-cache}_i$ \label{line:lin3} \Comment{returns $C_k$ or $A_k$}
  	 }\EndPart		
% 	
	}
	\end{multicols}
  \end{algorithmic}
\end{algorithm}
